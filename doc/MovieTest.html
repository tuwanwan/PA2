def initialize(data)
    @data = data
    @mean = 0
    @stddev = 0
    @rms = 0
    @data.each do |u, m, r, p|
        @mean += (p - r).abs
        @rms += (p - r)**2
    end
    @mean /= @data.length
    @rms = Math::sqrt(@rms / @data.length)
    @data.each do |u, m, r, p|
        @stddev += ((p - r).abs - @mean)**2
    end
    @stddev = Math::sqrt(@stddev / @data.length)
end
